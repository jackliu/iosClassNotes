IOS 中的多视图控制器

简介

在iOS中，同一个窗口中显示多个 View ，可以使用 UITabBar 的方式， 它由专门的控制器类 UITabBarController 控制，它与 UITabBarControllerDelegate 类有关联 。

关系

  • NSObject
    • UIResponder
      • UIViewController
        • UITabBarController

何处使用

  • 需要在屏幕底部划分模块时使用 UITabBar
  • UITabBar 独立于其他视图，单独由 UITabBarController 控制

新建视图.

  1. 鼠标定位到 左侧 与项目同名的 文件组上
  2. 右键-》New File
  3. 选择Cocoa Touch 栏目下的 Objective-C Class 点击next

承接上面.

  1. 填好类名 这里我写的是 LLViewController
  2. 父类 subclass of 为 UIViewController
  3. 不需要勾选创建对应 xib 文件的复选框
  4. 选择好要存放文件的文件夹 Create
  5. 重复上述操作，创建 MMViewController

程序启动调用类 xxxAppDelegate.m

xxxAppDelegate.h 
		
#import < UIKit/UIKit.h >

@interface AppDelegate : UIResponder < UIApplicationDelegate >
//程序启动，默认的窗口
@property (strong, nonatomic) UIWindow *window;

@end
xxxAppDelegate.m 
		
#import "AppDelegate.h"

#import "LLViewController.h"
#import "MMViewController.h"

@implementation AppDelegate

- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions
{
    self.window = [[UIWindow alloc] initWithFrame:[[UIScreen mainScreen] bounds]];

    //*******************************************************************************
    //这里我们学习 UITabBarViewController 的使用
    //*******************************************************************************
   
    //创建两个子视图控制器对象
    LLViewController *LLVC = [[LLViewController alloc] init];
    MMViewController *MMVC = [[MMViewController alloc] init];
    
    //设置LLVC MMVC 各自在第几个Tab上 和tab的图标
    //*******************************************************************************
    /*
    typedef enum {
   		UITabBarSystemItemMore,          //...
   		UITabBarSystemItemFavorites,     //星星
   		UITabBarSystemItemFeatured,      //探照灯
   		UITabBarSystemItemTopRated,      //星星
   		UITabBarSystemItemRecents,       //时钟
   		UITabBarSystemItemContacts,      //人头像
   		UITabBarSystemItemHistory,       //时钟
   		UITabBarSystemItemBookmarks,     //书
   		UITabBarSystemItemSearch,        //放大镜 
   		UITabBarSystemItemDownloads,     //下载
   		UITabBarSystemItemMostRecent,    //多本书
   		UITabBarSystemItemMostViewed,    //多个人
	} UITabBarSystemItem;
    */    
    //使用系统自带的图标，作为 TabBarItem 的图标
    //LLVC.tabBarItem = [[UITabBarItem alloc] initWithTabBarSystemItem:UITabBarSystemItemContacts tag:0];
    //MMVC.tabBarItem = [[UITabBarItem alloc] initWithTabBarSystemItem:UITabBarSystemItemBookmarks tag:1];
    //使用自定义的图标，设置 TabBarItem 图标，要求 30*30像素
    UIImage *img1 = [UIImage imageNamed:@"21-skull.png"];
    UIImage *img2 = [UIImage imageNamed:@"23-bird.png"];
    LLVC.tabBarItem = [[UITabBarItem alloc]initWithTitle:@"联系人" image:img1 tag:0];
    MMVC.tabBarItem = [[UITabBarItem alloc]initWithTitle:@"书签" image:img2 tag:1];
    //*******************************************************************************
    //把LLVC MMVC 设置成LLNav MMNav 的根VC，这样就可以在LLVC MMVC中使用pushViewController的方法了
    //如果使用下面代码，不可以使用导航方式跳转ViewController的
    //self.window.rootViewController = self.viewController;
    //*******************************************************************************
    UINavigationController *LLNav = [[UINavigationController alloc] initWithRootViewController:LLVC];
    UINavigationController *MMNav = [[UINavigationController alloc] initWithRootViewController:MMVC];
    
    //UITabBarController 可以控制多个 UINavigationController 所以使用数组
    //把LLNav MMNav生成一个数组，赋值给tabbarVC.viewControllers
    UITabBarController *tabbarVC = [[UITabBarController alloc] init];
    tabbarVC.viewControllers = [NSArray arrayWithObjects:LLNav, MMNav, nil];
    
    //设置tabbarVC 为winddow的rootViewController
    self.window.rootViewController = tabbarVC;
    
    self.window.backgroundColor = [UIColor whiteColor];
    [self.window makeKeyAndVisible];
    return YES;
}

对应实现代码 
    
- (void)applicationDidFinishLaunching:(UIApplication *)application {
   tabBarController = [[UITabBarController alloc] init];
 
   MyViewController* vc1 = [[MyViewController alloc] init];
   MyOtherViewController* vc2 = [[MyOtherViewController alloc] init];
 
   NSArray* controllers = [NSArray arrayWithObjects:vc1, vc2, nil];
   tabBarController.viewControllers = controllers;
 
    window.rootViewController = tabBarController;
}
